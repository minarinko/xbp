キー入力時のイベント処理 def type_event(self, event): self.ans_label2["text"] += event.keysym if __name__ == "__main__": root = tk.Tk() Application(master=root) root.mainloop() import random import time # 単語リスト word_list = ["りんご","ばなな","いちご","れもん","さくらんぼ","みかん","ぱいなっぷる","いちじく","もも","どらごんふるーつ","ぱぱいや","めろん","ますかっと","ぐあば","ぶるーべりー"] def random_word(): return random.choice(word_list) def main(): score = 0 game_duration = 30 #
