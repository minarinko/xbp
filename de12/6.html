自分に合うタイプの人診断



ソースコード

  1. questions = [
  2.     "1. どの季節が好きですか？\nA. 春\nB. 夏\nC. 秋\nD. 冬",
  3.     "2. どの種類の音楽が好きですか？\nA. ロック\nB. ポップ\nC. ジャズ\nD. クラシック",
  4.     "3. どのようなアクティビティが好きですか？\nA. アウトドア\nB. インドア\nC. スポーツ\nD. 読書",
  5.     "4.　占い師から告げられたあなたの前世は？\nA. 幸せに暮らした農民\nB.　歴史に名を遺した偉人\nC.　恋多き西洋の姫君\nD. 人ではなく動物だった",
  6. ]
  8. criteria = {
  9.     "A": "自然を楽しむタイプ",
  10.     "B": "社交的なタイプ",
  11.     "C": "クリエイティブなタイプ",
  12.     "D": "知的なタイプ",
  13. }
  15. # 質問に答える関数
  16. def ask_questions(questions):
  17.     answers = []
  18.     for question in questions:
  19.         print(question)
  20.         answer = input("選択肢を入力してください（A/B/C/D）: ")
  21.         answers.append(answer)
  22.     return answers
  24. # タイプを診断する関数
  25. def diagnose_person(answers, criteria):
  26.     result = {}
  27.     for answer in answers:
  28.         if answer in criteria:
  29.             result[criteria[answer]] = result.get(criteria[answer], 0) + 1
  30.     return result
  32. # メインプログラム
  33. if __name__ == "__main__":
  34.     print("自分に合うタイプの人を診断しましょう！")
  35.     user_answers = ask_questions(questions)
  36.     user_type = diagnose_person(user_answers, criteria)
  37.     
  38.     if user_type:
  39.         print("あなたに合うタイプの人は:")
  40.         for typ, count in user_type.items():
  41.             print(f"{typ}: {count}人")
  42.     else:
  43.         print("該当するタイプの人はいません。")
